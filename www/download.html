quick start
$ chmod +x n++
$ ./n++ hello.npp                          # compile → ./hello
$ ./n++ hello.npp -o my_program             # custom output name
$ ./n++ sieve.npp -O3                       # aggressive optimization
$ ./n++ app.npp -o release -O2             # balanced opt + named output
$ ./n++ -w myfile.npp                        # open in interactive editor
$ ./n++ --version                             # Nova Compiler - Beta 1.3.0
